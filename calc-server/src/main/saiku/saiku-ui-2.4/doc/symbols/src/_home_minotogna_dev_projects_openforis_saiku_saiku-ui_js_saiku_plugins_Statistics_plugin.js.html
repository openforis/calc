 22 /**
 23  * Renders a stats for each workspace
 24  */
 25 var Statistics = Backbone.View.extend({
 26     initialize: function(args) {
 27         this.workspace = args.workspace;
 28         
 29         // Create a unique ID for use as the CSS selector
 30         this.id = _.uniqueId("stats_");
 31         $(this.el).attr({ id: this.id });
 32         
 33         // Bind table rendering to query result event
 34         _.bindAll(this, "render", "receive_data", "process_data", "show", 
 35             "setOptions");
 36         this.workspace.bind('query:result', this.receive_data);
 37         
 38         // Add stats button
 39         this.add_button();
 40         this.workspace.toolbar.stats = this.show;
 41         
 42         // Listen to adjust event and rerender stats
 43         this.workspace.bind('workspace:adjust', this.render);
 44         
 45         // Append stats to workspace
 46         $(this.workspace.el).find('.workspace_results')
 47             .prepend($(this.el).hide())
 48     },
 49     
 50     add_button: function() {
 51         var $stats_button = 
 52             $('<a href="#stats" class="stats button disabled_toolbar i18n" title="Basic Statistics"></a>')
 53             .css({  'background-image': "url('js/saiku/plugins/Statistics/sigma.png')",
 54                     'background-repeat':'no-repeat',
 55                     'background-position':'50% 50%'
 56                 });
 57 
 58         var $stats_li = $('<li class="seperator"></li>').append($stats_button);
 59         $(this.workspace.toolbar.el).find("ul").append($stats_li);
 60     },
 61     
 62     show: function(event, ui) {
 63         $(this.workspace.el).find('.workspace_results table').toggle();
 64         $(this.el).toggle();
 65         $(event.target).toggleClass('on');
 66         
 67         if ($(event.target).hasClass('on')) {
 68             this.render();
 69         }
 70     },
 71     
 72     setOptions: function(event) {
 73         var type = $(event.target).attr('href').replace('#', '');
 74         try {
 75             this[type]();
 76         } catch (e) { }
 77         
 78         return false;
 79     },
 80     
 81     render: function() {
 82         if (! $(this.workspace.toolbar.el).find('.stats').hasClass('on')) {
 83             return;
 84         }
 85 
 86         var createRow = function(cells, header){
 87             var row = "<tr>"
 88             var ohcell = header?"<th class='row_header'>":"<th class='row'>"
 89             var odcell = header?"<th class='col'>":"<td class='data'>"
 90             var ccell = header?"</th>":"</td>"
 91             _.each(cells, function(it, idx){row += ((idx==0)?ohcell:odcell) 
 92 					+ "<div " + ((idx==0)?'class="i18n"':'') + ">"
 93 					+ it 
 94 					+ "</div>" + ccell})
 95             row += "</tr>"
 96             return row
 97         }
 98 
 99         var group = function(grid, el, cback){
100             var elements = _.filter(_.map(grid, function(it){return it[el]}), function(it){return it})
101             return cback(elements).toFixed(3)
102         }
103 
104         var sum = function(elems){return _.reduce(elems, function(memo, num){ return memo + num }, 0)}
105         var average = function(elems){return (sum(elems))/elems.length}
106         var stdx = function(elems){
107             var m = average(elems)
108             return Math.sqrt(sum(_.map(elems, function(it){return Math.pow(m - it,2)}))/elems.length)
109         }
110         var min = function(elems){return _.min(elems)}
111         var max = function(elems){return _.max(elems)}
112         
113         $(this.el).empty()
114         var grid = this.data.metadata
115         var rs = this.data.resultset
116         var aux = _.filter(grid, function(it){return !it.isHeader && it.colType == 'Numeric'})
117         var columns = _.map(aux, function(it){return it.colName})
118         var idxs =_.map(columns, function(el){return _.indexOf(_.map(grid, function(it){return it.colName}), el)})
119         
120         var $table = $("<table style='display: table; '>")
121         var $tbody = $("<tbody>").appendTo($table)
122         $tbody.append(createRow(['Statistics'].concat(columns), true))
123         $tbody.append(createRow(['Min'].concat(_.map(idxs, function(it){return group(rs, it, min)})), false))
124         $tbody.append(createRow(['Max'].concat(_.map(idxs, function(it){return group(rs, it, max)})), false))
125         $tbody.append(createRow(['Sum'].concat(_.map(idxs, function(it){return group(rs, it, sum)})), false))
126         $tbody.append(createRow(['Average'].concat(_.map(idxs, function(it){return group(rs, it, average)})), false))
127         $tbody.append(createRow(['Std. Deviation'].concat(_.map(idxs, function(it){return group(rs, it, stdx)})), false))
128 
129         $(this.el).append($table)
130         
131 		Saiku.i18n.translate();
132     },
133     
134     receive_data: function(args) {
135         return _.delay(this.process_data, 0, args);
136     },
137     
138     process_data: function(args) {
139         this.data = {};
140         this.data.resultset = [];
141         this.data.metadata = [];
142         this.data.height = 0;
143         this.data.width = 0;
144 
145         if (args.data.cellset && args.data.cellset.length > 0) {
146             
147             var lowest_level = 0;
148             var isHead = true
149             var columnNames = new Array()
150             for (var row = 0; row < args.data.cellset.length; row++) {
151                 if (isHead && (args.data.cellset[row][0].type == "ROW_HEADER_HEADER" || 
152                     args.data.cellset[row][0].value == "null")) {
153                     this.data.metadata = [];
154                     for (var field = 0; field < args.data.cellset[row].length; field++) {
155                         if (args.data.cellset[row][field].type == "ROW_HEADER_HEADER") {
156                             this.data.metadata.shift();
157                             lowest_level = field;
158                         }
159                         if(columnNames[field]){
160                             columnNames[field].push(args.data.cellset[row][field].value)
161                         }else{
162                             columnNames[field] = [args.data.cellset[row][field].value]
163                         }
164                         if(args.data.cellset[row][0].type == "ROW_HEADER_HEADER"){
165                             this.data.metadata.push({
166                                 colIndex: field,
167                                 colType: typeof(args.data.cellset[row + 1][field].value) !== "number" &&
168                                     isNaN(args.data.cellset[row + 1][field].value
169                                     .replace(/[^a-zA-Z 0-9.]+/g,'')) ? "String" : "Numeric",
170                                 colName: columnNames[field].join(' / '),
171                                 isHeader: (args.data.cellset[row][field].type == "ROW_HEADER_HEADER")
172                             });    
173                         }
174                     }
175                 } else if (args.data.cellset[row][lowest_level].value !== "null" && args.data.cellset[row][lowest_level].value !== "") {
176                     isHead = false
177                     var record = [];
178                     this.data.width = args.data.cellset[row].length;
179                     for (var col = lowest_level; col < args.data.cellset[row].length; col++) {
180                         var value = args.data.cellset[row][col].value;
181                         // check if the resultset contains the raw value, if not try to parse the given value
182                         if (args.data.cellset[row][col].properties.raw && args.data.cellset[row][col].properties.raw !== "null")
183                         {
184                             value = parseFloat(args.data.cellset[row][col].properties.raw);
185                         } else if (typeof(args.data.cellset[row][col].value) !== "number" &&
186                             parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''))) 
187                         {
188                             value = parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''));
189                         }
190                         if (col == lowest_level) {
191                             value += " [" + row + "]";
192                         }
193                         record.push(value);
194                     }
195                     this.data.resultset.push(record);
196                 }
197             }
198             this.data.height = this.data.resultset.length;
199             this.render();
200         } else {
201             $(this.el).text("No results");
202         }
203     }
204 });

206 /**
207  * Start Plugin
208  */ 
209  Saiku.events.bind('session:new', function(session) {
210 
211         function new_workspace(args) {
212             // Add stats element
213             if (typeof args.workspace.stats == "undefined") {
214                 args.workspace.stats = new Statistics({ workspace: args.workspace });
215             }
216         }
217 
218         function clear_workspace(args) {
219             if (typeof args.workspace.stats != "undefined") {
220                 $(args.workspace.stats.el).parents().find('.workspace_results table').show();
221                 $(args.workspace.stats.el).hide();
222             }
223         }
224 
225         
226         // Attach stats to existing tabs
227         for(var i = 0; i < Saiku.tabs._tabs.length; i++) {
228             var tab = Saiku.tabs._tabs[i];
229             new_workspace({
230                 workspace: tab.content
231             });
232         };
233 
234         // Attach stats to future tabs
235         Saiku.session.bind("workspace:new", new_workspace);
236         Saiku.session.bind("workspace:clear", clear_workspace);
237     });
