 33 var SavedQuery = Backbone.Model.extend({
 34 
 35     parse: function(response) {
 36         //console.log("response: " + response);
 37         //this.xml = response;
 38     },
 39     
 40     url: function() {
 41         var u = Settings.BIPLUGIN ? 
 42                 encodeURI(Saiku.session.username + "/pentahorepository2/resource")  
 43                     : encodeURI(Saiku.session.username + "/repository2/resource");
 44         return u;
 45     },
 46     
 47     move_query_to_workspace: function(model, response) {
 48         var file = response;
 49         var filename = model.get('file');
 50         for (var key in Settings) {
 51             if (key.match("^PARAM")=="PARAM") {
 52                 var variable = key.substring("PARAM".length, key.length)
 53                 file = file.replace("${" + variable + "}", Settings[key]);
 54             }
 55         }
 56         var query = new Query({ 
 57             xml: file,
 58             formatter: Settings.CELLSET_FORMATTER
 59         },{
 60             name: filename
 61         });
 62         
 63         var tab = Saiku.tabs.add(new Workspace({ query: query }));
 64     }
 65 });

 67 /**
 68  * Repository adapter
 69  */
 70 var Repository = Backbone.Collection.extend({
 71     model: SavedQuery,
 72     
 73     initialize: function(args, options) {
 74         this.dialog = options.dialog;
 75     },
 76     
 77     parse: function(response) {
 78         this.dialog.populate(response);
 79     },
 80     
 81     url: function() {
 82         var segment = Settings.BIPLUGIN ? 
 83             "/pentahorepository2/?type=saiku" : "/repository2/?type=saiku";
 84         return encodeURI(Saiku.session.username + segment);
 85     }
 86 });
