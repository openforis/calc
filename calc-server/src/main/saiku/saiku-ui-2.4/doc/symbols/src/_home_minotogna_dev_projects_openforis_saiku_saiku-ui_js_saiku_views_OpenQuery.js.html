 21 /**
 22  * The open query tab (Repository viewer)
 23  */
 24 var OpenQuery = Backbone.View.extend({
 25     className: 'tab_container',
 26     
 27     events: {
 28         'click .query': 'view_query',
 29         'dblclick .query': 'select_and_open_query',
 30         'click .add_folder' : 'add_folder',
 31         'click li.folder': 'toggle_folder',
 32         'click .workspace_toolbar a.open': 'open_query',
 33         'click .workspace_toolbar [href=#edit_folder]': 'edit_folder',
 34         'click .workspace_toolbar [href=#delete_folder]': 'delete_folder',
 35         'click .workspace_toolbar [href=#delete_query]': 'delete_query',
 36         'click .queries' : 'click_canvas'
 37     },
 38     
 39     template: function() {
 40         return _.template($("#template-open-dialog").html())();        
 41     },
 42 
 43     template_repository_objects: function( repository ) {
 44         var self = this;
 45         $(this.el).find('.sidebar ul').html(
 46             _.template( $( '#template-repository-objects' ).html( ) )( {
 47                 repoObjects: repository
 48             } ) 
 49         );
 50     },
 51     
 52     caption: function() {
 53         return "Repository";
 54     },
 55     
 56     render: function() {
 57         // Load template
 58         $(this.el).html(this.template());
 59         
 60         // Adjust tab when selected
 61         this.tab.bind('tab:select', this.fetch_queries);
 62         this.tab.bind('tab:select', this.adjust);
 63         $(window).resize(this.adjust);
 64         
 65         return this; 
 66     },
 67     
 68     initialize: function(args) {
 69         // Maintain `this`
 70         _.bindAll(this, "adjust", "fetch_queries",
 71                 "clear_query","select_and_open_query");
 72         
 73         // Initialize repository
 74         this.repository = new Repository({}, { dialog: this });
 75     },
 76     
 77     fetch_queries: function() {
 78         this.repository.fetch();
 79     },
 80     
 81     populate: function( repository ) {
 82         var self = this;
 83         self.template_repository_objects( repository );
 84         self.queries = {};
 85         function getQueries( entries ) {
 86             _.forEach( entries, function( entry ) {
 87                 if( entry.type === 'FILE' ) {
 88                     self.queries[ entry.path ] = entry;
 89                 } else {
 90                     getQueries( entry.repoObjects );
 91                 }
 92             } );
 93         }
 94         getQueries( repository );
 95     },
 96     
 97     view_query: function(event) {
 98         event.preventDefault( );
 99         var $currentTarget = $( event.currentTarget );
100         var $target = $currentTarget.find('a');
101         this.unselect_current_selected( );
102         $currentTarget.addClass( 'selected' );
103         var path = $target.attr('href').replace('#', '');
104         var name = $target.text();
105         var query = this.queries[path];
106         
107         $(this.el).find('.workspace_toolbar').removeClass( 'hide' );
108         $( this.el ).find( '.for_folder' ).addClass( 'hide' );
109         $( this.el ).find( '.for_queries' ).removeClass( 'hide' );
110         
111         var $results = $(this.el).find('.workspace_results')
112             .html('<h3><strong>' + query.name + '</strong></h3>');
113         var $properties = $('<ul id="query_info" />').appendTo($results);
114         
115         // Iterate through properties and show a key=>value set in the information pane
116         for (var property in query) {
117             if (query.hasOwnProperty(property) && property != "name") {
118                 $properties.append($('<li />').html("<strong>" + 
119                         property + "</strong> : " + query[property]));
120             }
121         }
122         
123         this.selected_query = new SavedQuery({ file: path, name: name });
124         
125         return false;
126     },
127 
128     view_folder: function( event ) {
129         var $target = $( event.currentTarget ).find( 'a' );
130         var name = $target.attr( 'href' ).replace( '#', '' );
131 
132         $( this.el ).find( '.workspace_toolbar' ).removeClass( 'hide' )
133         $( this.el ).find( '.for_queries' ).addClass( 'hide' );
134         $( this.el ).find( '.for_folder' ).removeClass( 'hide' );
135 
136         $( this.el ).find( '.workspace_results' )
137             .html( '<h3><strong>' + name + '</strong></h3>' );
138 
139         this.selected_query = new SavedQuery({ file: name , name: name });
140 
141     },
142 
143     add_folder: function( event ) {
144         $selected = $(this.el).find('.selected');
145         var path ="";
146         if (typeof $selected !== "undefined" && $selected) {
147             if ($selected.hasClass('folder_row')) {
148                 path = $selected.find('.folder_row a').attr('href');
149                 path = path.length > 1 ? path.substring(1,path.length) : path; 
150                 path+= "/";
151 
152             } else if ($selected.hasClass('query') && !$selected.parent().hasClass('RepositoryObjects')) {
153                 var query = $selected.find('a');
154                 path = query.attr('href');
155                 var queryname = query.text();
156                 path = path.substring(1, path.length - queryname.length );
157             }
158         }
159         
160         (new AddFolderModal({ 
161             path: path,
162             success: this.clear_query 
163         })).render().open();
164 
165         return false;
166     },
167 
168     click_canvas: function(event) {
169         var $target = $( event.currentTarget );
170         if ($target.hasClass('sidebar')) {
171             $(this.el).find('.selected').removeClass('selected');
172         }
173         return false;
174     },
175 
176     toggle_folder: function( event ) {
177         var $target = $( event.currentTarget );
178         this.unselect_current_selected( );
179         $target.children('.folder_row').addClass( 'selected' );
180         var $queries = $target.children( '.folder_content' );
181         var isClosed = $target.children( '.folder_row' ).find('.sprite').hasClass( 'collapsed' );
182         if( isClosed ) {
183             $target.children( '.folder_row' ).find('.sprite').removeClass( 'collapsed' );
184             $queries.removeClass( 'hide' );
185         } else {
186             $target.children( '.folder_row' ).find('.sprite').addClass( 'collapsed' );
187             $queries.addClass( 'hide' );
188         }
189 
190         this.view_folder( event );
191 
192         return false;
193     },
194 
195     select_and_open_query: function(event) {
196         var $target = $(event.currentTarget).find('a');
197         var path = $target.attr('href').replace('#', '');
198         var name = $target.text();
199         this.selected_query = new SavedQuery({ file: path, name: path });
200         this.open_query();
201     },
202     
203     open_query: function(event) {
204         Saiku.ui.block("Opening query...");
205         this.selected_query.fetch({ 
206             success: this.selected_query.move_query_to_workspace,
207             error: function() { Saiku.ui.unblock();},
208             dataType: "text"
209         });
210         
211         return false;
212     },
213 
214     delete_query: function(event) {
215         (new DeleteRepositoryObject({
216             query: this.selected_query,
217             success: this.clear_query
218         })).render().open();
219         
220         return false;
221     },
222     
223     edit_folder: function( event ) {
224         alert( 'todo: edit folder properties/permissions' );
225         return false;
226     },
227     
228     delete_folder: function( event ) {
229         (new DeleteRepositoryObject({
230             query: this.selected_query,
231             success: this.clear_query
232         })).render().open();
233         return false;
234     },
235 
236     clear_query: function() {
237         $(this.el).find('.workspace_toolbar').addClass('hide');
238         $(this.el).find('.workspace_results').html('');
239         this.fetch_queries();
240     },
241     
242     adjust: function() {
243         // Adjust the height of the separator
244         $separator = $(this.el).find('.sidebar_separator');
245         $separator.height($("body").height() - 87);
246         $(this.el).find('.sidebar').height($("body").height() - 87);
247         
248         // Adjust the dimensions of the results window
249         $(this.el).find('.workspace_results').css({
250             width: $(document).width() - $(this.el).find('.sidebar').width() - 30,
251             height: $(document).height() - $("#header").height() -
252                 $(this.el).find('.workspace_toolbar').height() - 
253                 $(this.el).find('.workspace_fields').height() - 40
254         });
255     },
256     
257     toggle_sidebar: function() {
258         // Toggle sidebar
259         $(this.el).find('.sidebar').toggleClass('hide');
260         var new_margin = $(this.el).find('.sidebar').hasClass('hide') ?
261                 5 : 265;
262         $(this.el).find('.workspace_inner').css({ 'margin-left': new_margin });
263     },
264 
265     unselect_current_selected: function( ) {
266         $( this.el ).find( '.selected' ).removeClass( 'selected' );
267     }
268 
269 });
