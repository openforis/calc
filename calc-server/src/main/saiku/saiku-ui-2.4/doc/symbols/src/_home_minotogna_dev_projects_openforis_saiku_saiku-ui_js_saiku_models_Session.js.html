 21 /**
 22  * Object which handles authentication and stores connections and cubes
 23  * @param username
 24  * @param password
 25  * @returns {Session}
 26  */
 27 var Session = Backbone.Model.extend({
 28     username: null,
 29     password: null,
 30     sessionid: null,
 31         
 32     initialize: function(args, options) {
 33         // Attach a custom event bus to this model
 34         _.extend(this, Backbone.Events);
 35         _.bindAll(this, "check_session", "process_session", "load_session","login");
 36         // Check if credentials are being injected into session
 37         if (options && options.username && options.password) {
 38             this.username = options.username;
 39             this.password = options.password;
 40             this.save({username:this.username, password:this.password},{success: this.check_session, error: this.check_session});
 41 
 42         } else {
 43             this.check_session();
 44         }
 45     },
 46 
 47     check_session: function() {
 48         if (this.sessionid === null || this.username === null || this.password === null) {
 49             this.clear();
 50             this.fetch({ success: this.process_session })
 51         } else {
 52             this.username = encodeURIComponent(options.username);
 53             this.load_session();
 54         }
 55     },
 56 
 57     load_session: function() {
 58         this.sessionworkspace = new SessionWorkspace();
 59     },
 60 
 61     process_session: function(model, response) {
 62         if ((response === null || response.sessionid == null)) {
 63             // Open form and retrieve credentials
 64             Saiku.ui.unblock();
 65             this.form = new LoginForm({ session: this });
 66             this.form.render().open();
 67         } else {
 68             this.sessionid = response.sessionid;
 69             this.roles = response.roles;
 70             this.username = encodeURIComponent(response.username);
 71             this.load_session();
 72         }
 73         
 74         return this;
 75     },
 76     
 77     error: function() {
 78         $(this.form.el).dialog('open');
 79     },
 80     
 81     login: function(username, password) {
 82         // Set expiration on localStorage to one day in the future
 83         var expires = (new Date()).getTime() + 
 84             Settings.LOCALSTORAGE_EXPIRATION;
 85         typeof localStorage !== "undefined" && localStorage && localStorage.setItem('expiration', expires);
 86 
 87         this.save({username:username, password:password},{success: this.check_session, error: this.check_session});
 88         
 89     },
 90     
 91     logout: function() {
 92         // FIXME - This is a hack (inherited from old UI)
 93         Saiku.ui.unblock();
 94         $('#header').empty().hide();
 95         $('#tab_panel').remove();
 96 
 97         Saiku.toolbar = new Toolbar();
 98         typeof localStorage !== "undefined" && localStorage && localStorage.clear();
 99         this.id = _.uniqueId('queryaction_');
100         this.clear();
101         this.sessionid = null;
102         this.username = null;
103         this.password = null;
104         this.destroy({success:this.check_session, error:this.check_session});
105         delete this.id;
106 
107     },
108 
109     url: function() {
110 
111         return "session";
112     }
113 });
