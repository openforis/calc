 21 /**
 22  * Workspace query
 23  */
 24 var Query = Backbone.Model.extend({
 25 
 26     initialize: function(args, options) {
 27         // Save cube
 28         _.extend(this, options);
 29         
 30         // Bind `this`
 31         _.bindAll(this, "run", "move_dimension", "reflect_properties");
 32         
 33         // Generate a unique query id
 34         this.uuid = 'xxxxxxxx-xxxx-xxxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, 
 35             function (c) {
 36                 var r = Math.random() * 16 | 0,
 37                 v = c == 'x' ? r : (r & 0x3 | 0x8);
 38                 return v.toString(16);
 39             }).toUpperCase();
 40         
 41         // Initialize properties, action handler, and result handler
 42         this.action = new QueryAction({}, { query: this });
 43         this.result = new Result({}, { query: this });
 44         this.scenario = new QueryScenario({}, { query: this });
 45         this.set({type:'QM'});
 46     },
 47     
 48     parse: function(response) {
 49         // Assign id so Backbone knows to PUT instead of POST
 50         this.id = this.uuid;
 51 
 52         this.set({
 53             connection: response.cube.connectionName,
 54                 catalog: response.cube.catalogName,
 55                 schema: response.cube.schemaName,
 56                 cube: encodeURIComponent(response.cube.name),
 57                 axes: response.saikuAxes,
 58                 type: response.type
 59         })
 60 
 61         // Fetch initial properties from server
 62         if (! this.properties) {
 63             this.properties = new Properties({}, { query: this });
 64         } else {
 65             this.properties.fetch({
 66                 success: this.reflect_properties
 67             });
 68         }
 69     },
 70     
 71     reflect_properties: function() {
 72         this.workspace.trigger('properties:loaded');
 73     },
 74     
 75     run: function(force, mdx) {
 76         // Check for automatic execution
 77 
 78         
 79         if ( (this.properties.properties['saiku.olap.query.automatic_execution'] == "false") &&
 80             ! (force === true)) {
 81             return;
 82         }
 83         Saiku.ui.unblock();
 84         $(this.workspace.el).find(".workspace_results_info").empty();
 85 
 86         // TODO - Validate query
 87         // maybe we should sync it with the backend query JSON?
 88         // this definitely needs improvement
 89         var rows = 0;
 90         var columns = 0;
 91         if (this.get('type') != "MDX") {
 92             if (Settings.MODE == "view" || Settings.MODE == "table") {
 93                 var axes = this.get('axes');
 94                 if (axes) {
 95                     for (var axis_iter = 0; axis_iter < axes.length; axis_iter++) {
 96                         var axis = axes[axis_iter];
 97                         if (axis.name && axis.name == "ROWS") {
 98                             rows = axis.dimensionSelections.length;
 99                         }
100                         if (axis.name && axis.name == "COLUMNS") {
101                             columns = axis.dimensionSelections.length;
102                         }
103                     }
104                 }
105             } else {
106                 rows = $(this.workspace.el).find('.rows ul li').size();
107                 columns = $(this.workspace.el).find('.columns ul li').size(); 
108             }
109 
110             if (rows == 0 || columns == 0) {
111                 $(this.workspace.el).find('.workspace_results table')
112                     .html('<tr><td><span class="i18n">You need to put at least one level or measure on Columns and Rows for a valid query.</td></tr>');
113 				Saiku.i18n.translate();
114                 return;
115             }
116         }
117 
118         // Run it
119         $(this.workspace.table.el)
120             .html('<tr><td><span class="i18n">Running query...</span></td></tr>');
121 		Saiku.i18n.translate();
122             // <a class="cancel" href="#cancel">x</a>
123 
124         if (this.get('type')  == "MDX" && mdx != null) {
125             this.result.save({ mdx: mdx });
126         } else {
127             this.result.fetch();
128         }
129     },
130     
131     move_dimension: function(dimension, target, index) {
132         $(this.workspace.el).find('.run').removeClass('disabled_toolbar');
133         var url = "/axis/" + target + "/dimension/" + dimension;
134         
135         this.action.post(url, {
136             data: {
137                 position: index
138             },
139             
140             success: function() {
141                 if (this.query.properties
142                     .properties['saiku.olap.query.automatic_execution'] === 'true') {
143                     this.query.run();
144                 }
145             }
146         });
147     },
148     
149     url: function() {
150         return encodeURI(Saiku.session.username + "/query/" + this.uuid);
151     }
152 });
