 24 ;(function (window) {
 25 
 26   var
 27     characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=',
 28     fromCharCode = String.fromCharCode,
 29     INVALID_CHARACTER_ERR = (function () {
 30       // fabricate a suitable error object
 31       try { document.createElement('$'); }
 32       catch (error) { return error; }}());
 33 
 34   // encoder
 35   window.Base64 || (
 36   window.Base64 = { encode: function (string) {
 37     var
 38       a, b, b1, b2, b3, b4, c, i = 0,
 39       len = string.length, max = Math.max, result = '';
 40 
 41     while (i < len) {
 42       a = string.charCodeAt(i++) || 0;
 43       b = string.charCodeAt(i++) || 0;
 44       c = string.charCodeAt(i++) || 0;
 45 
 46       if (max(a, b, c) > 0xFF) {
 47         throw INVALID_CHARACTER_ERR;
 48       }
 49 
 50       b1 = (a >> 2) & 0x3F;
 51       b2 = ((a & 0x3) << 4) | ((b >> 4) & 0xF);
 52       b3 = ((b & 0xF) << 2) | ((c >> 6) & 0x3);
 53       b4 = c & 0x3F;
 54 
 55       if (!b) {
 56         b3 = b4 = 64;
 57       } else if (!c) {
 58         b4 = 64;
 59       }
 60       result += characters.charAt(b1) + characters.charAt(b2) + characters.charAt(b3) + characters.charAt(b4);
 61     }
 62     return result;
 63   }});

 67 /**
 68  * Model which handles AJAX calls to the Saiku Server
 69  * If you want to hook the UI up to something besides the Saiku Server,
 70  * this is the class which you want to override.
 71  * @returns {SaikuServer}
 72  */
 73 Backbone.sync = function(method, model, options) {
 74     var params;
 75     methodMap = {
 76         'create': "POST",
 77         'read': "GET",
 78         'update': "PUT",
 79         'delete': "DELETE"
 80     };
 81     
 82     // Generate AJAX action
 83     var type = methodMap[method];
 84     var url = Settings.REST_URL
 85         + (_.isFunction(model.url) ? model.url() : model.url);
 86     
 87     // Prepare for failure
 88     options.retries = 0;
 89     
 90     var statuscode = {
 91       0: function() {
 92         Saiku.session.logout();
 93       },
 94       401: function() {
 95         Saiku.session.logout();
 96       },
 97       403: function() {
 98         Saiku.session.logout();
 99       }
100 
101 
102     };
103 
104     var failure = function(jqXHR, textStatus, errorThrown) {
105       if (options.error) {
106                 options.error(jqXHR, textStatus, errorThrown);
107       }
108 /*      if (jqXHR.status)
109         options.retries++;
110         if (options.retries >= 10) {
111             Saiku.ui.block("Could not reach server. Please try again later...");
112             if (options.error) {
113                 options.error(jqXHR, textStatus, errorThrown);
114             }
115         } 
116          else {
117             var delay = Math.pow(options.retries, 2);
118             Saiku.ui.block("Having trouble reaching server. Trying again in " + delay + " seconds...");
119             setTimeout(function() {
120                 $.ajax(params);
121             }, delay * 1000);
122         } */
123     };
124 
125     var success = function(data, textStatus, jqXHR) {
126         if (options.retries > 0) {
127             Saiku.ui.unblock();
128         }
129         
130         options.success(data, textStatus, jqXHR);
131     };
132     var async = true
133     if (options.async === false) {
134       async = false;
135     }
136     var dataType = 'json'
137     if (typeof options.dataType != "undefined") {
138       dataType = options.dataType;
139     }
140     // Default JSON-request options.
141     params = {
142       url:          url,
143       type:         type,
144       cache:        false,
145       data:         model.attributes,
146       contentType:  'application/x-www-form-urlencoded',
147       dataType:     dataType,
148       success:      success,
149       statusCode:   statuscode, 
150       error:        failure,
151       async:        async
152       /*
153       beforeSend:   function(request) {
154         if (!Settings.PLUGIN) {
155           var auth = "Basic " + Base64.encode(
156               Saiku.session.username + ":" + Saiku.session.password
157           );
158           request.setRequestHeader('Authorization', auth);
159           }
160       } */
161     };
162 
163     // For older servers, emulate HTTP by mimicking the HTTP method with `_method`
164     // And an `X-HTTP-Method-Override` header.
165     if (Settings.BIPLUGIN || Backbone.emulateHTTP) {
166       if (type === 'PUT' || type === 'DELETE') {
167         if (Backbone.emulateHTTP) params.data._method = type;
168         params.type = 'POST';
169         params.beforeSend = function(xhr) {
170           xhr.setRequestHeader("X-HTTP-Method-Override", type);
171         };
172       }
173     }
174 
175     // Make the request
176     $.ajax(params);
177 };
