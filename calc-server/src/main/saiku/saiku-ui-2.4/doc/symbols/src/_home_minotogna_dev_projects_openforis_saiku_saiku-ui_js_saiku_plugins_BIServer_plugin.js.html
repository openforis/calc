 32 var puc = {
 33     allowSave: function(isAllowed) {
 34         if(window.parent.mantle_initialized !== undefined && window.parent.mantle_initialized && 
 35             window.parent.enableAdhocSave ) {
 36             if (window.ALLOW_PUC_SAVE === undefined || ALLOW_PUC_SAVE) {
 37                 window.parent.enableAdhocSave(isAllowed);
 38             }
 39         }
 40     },
 41 
 42     refresh_repo: function() {
 43         if(window.parent.mantle_initialized !== undefined && window.parent.mantle_initialized) {
 44             window.parent.mantle_refreshRepository();
 45         }
 46     },
 47 
 48     save_to_solution: function(filename, solution, path, type, overwrite) {
 49         var query = Saiku.tabs._tabs[0].content.query;
 50         query.action.get("/xml", {
 51             success: function(model, response) {
 52                     filename = filename 
 53                                 && filename.length > ".saiku".length
 54                                 && filename.substring(filename.length - ".saiku".length,filename.length) == ".saiku" ? filename : filename + ".saiku";
 55 
 56                     var file = (solution ? (solution + "/") : "")
 57                         + (path ? (path + "/") : "")
 58                         + (filename || "");
 59 
 60                     (new SavedQuery({
 61                         name: filename,
 62                         file: file,
 63                         content: response.xml
 64                 })).save({ success: function() {
 65                     puc.refresh_repo();
 66                 }});
 67             }
 68         });
 69     }
 70 };

105  * Bind callbacks to workspace
106  */
107 var BIPlugin = {
108     bind_callbacks: function(workspace) {
109         // If in view mode, remove sidebar and drop zones
110         if (Settings.MODE == "view" || Settings.MODE == "table") {
111             workspace.toggle_sidebar();
112             $(workspace.el).find('.sidebar_separator').remove();
113             $(workspace.el).find('.workspace_inner')
114                 .css({ 'margin-left': 0 });
115             $(workspace.el).find('.workspace_fields').remove();
116         }
117 
118         // Remove toolbar buttons
119         $(workspace.toolbar.el).find('.run').parent().removeClass('seperator');
120         if (Settings.MODE == "view" || Settings.MODE == "table") {
121             $(workspace.toolbar.el)
122                 .find(".run, .auto, .toggle_fields, .toggle_sidebar")
123                 .parent().remove();
124         }
125         if (Settings.MODE == "table") {
126             $(workspace.toolbar.el).parent().remove();
127         }
128 
129         // Toggle save button
130         workspace.bind('query:result', function(args) {
131             var isAllowed = args.data.cellset && 
132                 args.data.cellset.length > 0;
133             puc.allowSave(isAllowed);
134         });
135     }
136 };

149         }
150 
151         Saiku.session.bind('workspace:new', function(args) {
152             BIPlugin.bind_callbacks(args.workspace);
153         });
154     }

179     window.parent.getSaikuMdx = function() {
180             var myself = this;
181             var query = Saiku.tabs._tabs[0].content.query;
182             query.clear();
183             query.fetch({ 
184             success: function(model, response) {
185                     var ds = new Datasources();
186                     ds.fetch({
187                         success: function(dmodel, dresponse) {
188                             for (var i = 0; i < dresponse.length; i ++) {
189                                 if (dresponse[i].name == response.cube.connectionName) {
190                                     var urlParts = dresponse[i].properties.location.split(';');
191                                     var jndi = "";
192                                     var catalog = "";
193                                     $.each(urlParts,function(index, value) { 
194                                         var prop = value.split('=');
195                                         if (prop[0] == "DataSource") {
196                                             jndi = prop[1];
197                                         }
198                                        if (prop[0] == "Catalog") {
199                                             catalog = prop[1];
200                                         } 
201                                     });
202                                     var saikuStub = {
203                                         connection: dresponse[i].name,
204                                         catalog: catalog,
205                                         jndi: jndi,
206                                         mdx: response.mdx
207                                     }
208                                     window.parent.saveSaiku(saikuStub);
209                                 }        
210                             }
211                         }
212                     })
213                     
214                 }
215             });
216     };
