 22 /**
 23  * Controls the appearance and behavior of the dimension list
 24  * 
 25  * This is where drag and drop lives
 26  */
 27 var DimensionList = Backbone.View.extend({
 28     events: {
 29         'click span': 'select',
 30         'click a': 'select',
 31         'click .parent_dimension ul li a' : 'select_dimension',
 32         'click .measure' : 'select_measure'
 33     },
 34     
 35     initialize: function(args) {
 36         // Don't lose this
 37         _.bindAll(this, "render", "load_dimension","select_dimension");
 38         
 39         // Bind parent element
 40         this.workspace = args.workspace;
 41         this.dimension = args.dimension;
 42         
 43         // Fetch from the server if we haven't already
 44         if (args.dimension && args.dimension.has('template')) {
 45             this.load_dimension();
 46         } else if (! args.dimension){
 47             $(this.el).html('Could not load dimension. Please log out and log in again.');
 48         } else {
 49             $(this.el).html('Loading...');
 50             args.dimension.fetch({ success: this.load_dimension });
 51         }
 52     },
 53     
 54     load_dimension: function() {
 55         this.template = this.dimension.get('template');
 56         this.render(); 
 57         this.workspace.trigger('dimensions:loaded', $(this.el));
 58 
 59     },
 60     
 61     render: function() {
 62         // Pull the HTML from cache and hide all dimensions
 63         $(this.el).html(this.template)
 64             .find('.hide').hide().removeClass('hide');
 65         
 66         // Add draggable behavior
 67         $(this.el).find('.measure,.level').parent('li').draggable({
 68             cancel: '.not-draggable, .hierarchy',
 69             connectToSortable: $(this.workspace.el)
 70                 .find('.columns > ul, .rows > ul, .filter > ul'),
 71             helper: 'clone',
 72             opacity: 0.60,
 73             tolerance: 'pointer',
 74             cursorAt: {
 75                 top: 10,
 76                 left: 35
 77             }
 78         });
 79     },
 80     
 81     select: function(event) {
 82         var $target = $(event.target).hasClass('root')
 83             ? $(event.target) : $(event.target).parent().find('span');
 84         if ($target.hasClass('root')) {
 85             $target.find('a').toggleClass('folder_collapsed').toggleClass('folder_expand');
 86             $target.toggleClass('collapsed').toggleClass('expand');
 87             $target.parents('li').find('ul').children('li').toggle();
 88         }
 89         
 90         return false;
 91     },
 92 
 93      select_dimension: function(event, ui) {
 94         if ($(event.target).parent().hasClass('ui-state-disabled')) {
 95             return;
 96         }
 97         
 98         $axis = $(this.workspace.el).find(".columns ul li").length > 0 ?
 99             $(this.workspace.el).find(".rows ul") :
100             $(this.workspace.el).find(".columns ul");
101         $target = $(event.target).parent().clone()
102             .appendTo($axis);
103         this.workspace.drop_zones.select_dimension({
104             target: $axis
105         }, {
106             item: $target
107         });
108         return false;
109     },
110 
111     select_measure: function(event, ui) {
112         if ($(event.target).parent().hasClass('ui-state-disabled')) {
113             return;
114         }
115         
116         var $axis;
117         if ($(this.workspace.el).find(".rows ul .d_measure").length > 0) $axis = $(this.workspace.el).find(".rows ul");
118         else if ($(this.workspace.el).find(".columns ul .d_measure").length > 0) $axis = $(this.workspace.el).find(".columns ul");
119         else if ($(this.workspace.el).find(".filter ul .d_measure").length > 0) $axis = $(this.workspace.el).find(".filter ul");
120         else $axis = $(this.workspace.el).find(".columns ul");
121 
122         $target = $(event.target).parent().clone();
123         if ($axis.find(".d_measure").length != 0)
124             $target.insertAfter($axis.find(".d_measure:last"));
125         else {
126             $target.appendTo($axis);
127         }
128         this.workspace.drop_zones.select_dimension({
129             target: $axis
130         }, {
131             item: $target
132         });
133         return false;
134     }
135 });
