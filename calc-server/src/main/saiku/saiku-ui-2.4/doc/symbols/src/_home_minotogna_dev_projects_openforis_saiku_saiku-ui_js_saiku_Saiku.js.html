 21 /**
 22  * Central object for handling global application state
 23  */
 24 var Saiku = {
 25     /**
 26      * View which manages toolbar interactions
 27      */
 28     toolbar: {},
 29     
 30     /**
 31      * View which handles tabs
 32      */
 33     tabs: new TabSet(),
 34     
 35     /**
 36      * Model which handles session and authentication
 37      */
 38     session: null,
 39     
 40     /**
 41      * Global event bus
 42      */
 43     events: _.extend({}, Backbone.Events),
 44     
 45     /**
 46      * Collection of routers for page fragments
 47      */
 48     routers: [],
 49     
 50     /**
 51      * Convenience functions for blocking the UI
 52      */
 53     ui: {
 54         block: function(message) {
 55             $('.processing,.processing_container').fadeIn();
 56             $('.processing_message').text(message);
 57 			
 58 			$('.processing_message').removeClass("i18n_translated").addClass("i18n");
 59 			Saiku.i18n.translate();
 60         },
 61         
 62         unblock: function() {
 63             $('.processing,.processing_container').fadeOut();
 64         }
 65     }
 66 };
 67 
 68 /**
 69  * Setting this option to true will fake PUT and DELETE requests 
 70  * with a HTTP POST, and pass them under the _method parameter. 
 71  * Setting this option will also set an X-HTTP-Method-Override header 
 72  * with the true method. This is required for BI server integration
 73  */
 74 Backbone.emulateHTTP = false;
 75 
 76 /**
 77  * Up up and away!
 78  */
 79 if (! Settings.BIPLUGIN) {
 80     $(document).ready(function() {
 81         Saiku.session = new Session({}, {
 82             username: Settings.USERNAME,
 83             password: Settings.PASSWORD
 84         });
 85 
 86         Saiku.toolbar = new Toolbar();
 87     });
 88 }
