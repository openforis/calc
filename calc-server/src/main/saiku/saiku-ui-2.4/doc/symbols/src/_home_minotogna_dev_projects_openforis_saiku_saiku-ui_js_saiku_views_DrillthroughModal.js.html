1 /**
  2  * Dialog for member selections
  3  */
  4 var DrillthroughModal = Modal.extend({
  5     type: "drillthrough",
  6     
  7     buttons: [
  8         { text: "Ok", method: "ok" },
  9         { text: "Cancel", method: "close" }
 10     ],
 11 
 12     events: {
 13         'click .collapsed': 'select',
 14         'click .expand': 'select',
 15         'click .folder_collapsed': 'select',
 16         'click .folder_expanded': 'select',
 17         
 18         'click .dialog_footer a:' : 'call',
 19         'click .parent_dimension input' : 'select_dimension',
 20         'click .measure_tree input' : 'select_measure'
 21     },
 22 
 23     
 24     initialize: function(args) {
 25         // Initialize properties
 26         _.extend(this, args);
 27         this.options.title = args.title;
 28         this.query = args.workspace.query;
 29         
 30         this.position = args.position;
 31         this.action = args.action;
 32         Saiku.ui.unblock();
 33         _.bindAll(this, "ok", "drilled");
 34 
 35         // Resize when rendered
 36         this.bind('open', this.post_render);
 37         this.render();
 38                // Load template
 39        $(this.el).find('.dialog_body')
 40           .html(_.template($("#template-drillthrough").html())(this));
 41         // Show dialog
 42         $(this.el).find('.maxrows').val(this.maxrows);
 43                     
 44         var schema = this.query.get('schema');
 45         var key = this.query.get('connection') + "/" + 
 46                 this.query.get('catalog') + "/"
 47                 + ((schema == "" || schema == null) ? "null" : schema) 
 48                 + "/" + this.query.get('cube');
 49 
 50         var container = $("#template-drillthrough-list").html();
 51         var dimensions = Saiku.session.sessionworkspace.dimensions[key].get('data');
 52         var measures = Saiku.session.sessionworkspace.measures[key].get('data');
 53 
 54         if (typeof dimensions == "undefined" || typeof measures == "undefined") {
 55                         if (typeof localStorage !== "undefined" && localStorage && 
 56                             localStorage.getItem("dimension." + key) !== null &&
 57                             localStorage.getItem("measure." + key) !== null) {
 58                             Saiku.session.sessionworkspace.dimensions[key] = new Dimension(JSON.parse(localStorage.getItem("dimension." + key)));
 59                             Saiku.session.sessionworkspace.measures[key] = new Measure(JSON.parse(localStorage.getItem("measure." + key)));
 60                         } else {
 61                             Saiku.session.sessionworkspace.dimensions[key] = new Dimension({ key: key });
 62                             Saiku.session.sessionworkspace.measures[key] = new Measure({ key: key });
 63                             Saiku.session.sessionworkspace.dimensions[key].fetch({ async : false });
 64                             Saiku.session.sessionworkspace.measures[key].fetch({ async : false });
 65                         }
 66                         dimensions = Saiku.session.sessionworkspace.dimensions[key].get('data');
 67                         measures = Saiku.session.sessionworkspace.measures[key].get('data');
 68         } 
 69 
 70         var templ_dim =_.template($("#template-drillthrough-dimensions").html())({dimensions: dimensions});
 71         var templ_measure =_.template($("#template-drillthrough-measures").html())({measures: measures});
 72 
 73         $(container).appendTo($(this.el).find('.dialog_body'));
 74         $(this.el).find('.sidebar').height(($("body").height() / 2) + ($("body").height() / 6) );
 75         $(this.el).find('.sidebar').width(380);
 76 
 77         $(this.el).find('.dimension_tree').html('').append($(templ_dim));
 78         $(this.el).find('.measure_tree').html('').append($(templ_measure));
 79         
 80 		Saiku.i18n.translate();
 81     },
 82     
 83     select: function(event) {
 84         var $target = $(event.target).hasClass('root')
 85             ? $(event.target) : $(event.target).parent().find('span');
 86         if ($target.hasClass('root')) {
 87             $target.find('a').toggleClass('folder_collapsed').toggleClass('folder_expand');
 88             $target.toggleClass('collapsed').toggleClass('expand');
 89             $target.parents('li').find('ul').children('li').toggle();
 90         }
 91         
 92         return false;
 93     },
 94 
 95     select_dimension: function(event) {
 96         var $target = $(event.target);
 97         var checked = $target.is(':checked');
 98         $target.parent().find('input').attr('checked', checked);
 99     },
100 
101     select_measure: function(event) {
102         var $target = $(event.target);
103         var checked = $target.is(':checked');
104         if(checked) {
105             $target.parent().siblings().find('input').attr('checked', false);
106         }
107     },
108 
109 
110     post_render: function(args) {
111         $(args.modal.el).parents('.ui-dialog').css({ width: "150px" });
112     },
113     
114     ok: function() {
115         // Notify user that updates are in progress
116         var $loading = $("<div>Drilling through...</div>");
117         $(this.el).find('.dialog_body').children().hide();
118         $(this.el).find('.dialog_body').prepend($loading);
119         var selections = ""
120         $(this.el).find('.check_level:checked').each( function(index) { 
121             if (index > 0) {
122                 selections += ", ";
123             }
124             selections += $(this).val();
125         });
126 
127         var maxrows = $(this.el).find('.maxrows').val();
128         var params = "?maxrows=" + maxrows;
129         params = params + (typeof this.position !== "undefined" ? "&position=" + this.position : "" );
130         params += "&returns=" + selections;
131         if (this.action == "export") {
132             var location = Settings.REST_URL +
133                 Saiku.session.username + "/query/" + 
134                 this.query.id + "/drillthrough/export/csv" + params;
135             this.close();
136             window.open(location);
137         } else if (this.action == "table") {
138             Saiku.ui.block("Executing drillthrough...");
139             this.query.action.get("/drillthrough", { data: { position: this.position, maxrows: maxrows , returns: selections}, success: this.drilled } );
140             this.close();
141         }
142         
143         return false;
144     },
145 
146     drilled: function(model, response) {
147         var table = new Table({ workspace: this.workspace });
148         table.render({ data: response }, true);
149 
150         Saiku.ui.unblock();
151         var html = '<div id="fancy_results" class="workspace_results" style="overflow:visible"><table>' + $(table.el).html() + '</table></div>';
152         this.remove();
153         table.remove();
154         $.fancybox(html
155             ,
156             {
157             'autoDimensions'    : false,
158             'autoScale'         : false,
159             'height'            :  ($("body").height() - 100),
160             'width'             :  ($("body").width() - 100),
161             'transitionIn'      : 'none',
162             'transitionOut'     : 'none'
163             }
164         );
165 
166     },
167     
168     finished: function() {
169         $(this.el).dialog('destroy').remove();
170         this.query.run();
171     }
172 });
