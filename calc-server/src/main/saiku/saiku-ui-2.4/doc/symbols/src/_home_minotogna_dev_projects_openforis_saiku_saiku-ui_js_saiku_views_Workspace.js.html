 21 /**
 22  * The analysis workspace
 23  */
 24 var Workspace = Backbone.View.extend({
 25     className: 'tab_container',
 26     
 27     events: {
 28         'click .sidebar_separator': 'toggle_sidebar',
 29         'change .cubes': 'new_query',
 30         'drop': 'remove_dimension',
 31         'click .refresh_cubes' : 'refresh'
 32     },
 33     
 34     initialize: function(args) {
 35         // Maintain `this` in jQuery event handlers
 36         _.bindAll(this, "adjust", "toggle_sidebar", "prepare", "new_query", 
 37                 "init_query", "update_caption", "populate_selections","refresh");
 38                 
 39         // Attach an event bus to the workspace
 40         _.extend(this, Backbone.Events);
 41         this.loaded = false;
 42         this.bind('dimensions:loaded',this.populate_selections);
 43 
 44         // Generate toolbar and append to workspace
 45         this.toolbar = new WorkspaceToolbar({ workspace: this });
 46         this.toolbar.render();
 47         
 48         // Create drop zones
 49         this.drop_zones = new WorkspaceDropZone({ workspace: this });
 50         this.drop_zones.render();
 51         
 52         // Generate table
 53         this.table = new Table({ workspace: this });
 54         
 55         // Pull query from args
 56         if (args && args.query) {
 57             this.query = args.query;
 58             this.query.workspace = this;
 59             this.query.save({}, { success: this.init_query });            
 60         }
 61         // Flash cube navigation when rendered
 62         Saiku.session.bind('tab:add', this.prepare);
 63     },
 64     
 65     caption: function() {
 66         if (this.query && this.query.name) {
 67             return this.query.name;
 68         }
 69         
 70         return "Unsaved query (" + (Saiku.tabs.queryCount + 1) + ")";
 71     },
 72     
 73     template: function() {
 74         var template = $("#template-workspace").html() || "";
 75         return _.template(template)({
 76             cube_navigation: Saiku.session.sessionworkspace.cube_navigation
 77         });        
 78     },
 79 
 80     refresh: function() {
 81         Saiku.session.sessionworkspace.refresh();
 82     },
 83     
 84     render: function() {
 85         // Load template
 86         $(this.el).html(this.template());
 87         
 88         // Show toolbar
 89         $(this.el).find('.workspace_toolbar').append($(this.toolbar.el));
 90         
 91         // Show drop zones
 92         $(this.el).find('.workspace_editor').append($(this.drop_zones.el));
 93 
 94         
 95         // Activate sidebar for removing elements
 96         $(this.el).find('.sidebar')
 97             .droppable({
 98                 accept: '.d_measure, .d_dimension'
 99             });
100         
101         // Add results table
102         $(this.el).find('.workspace_results')
103             .append($(this.table.el));
104         
105         // Adjust tab when selected
106         this.tab.bind('tab:select', this.adjust);
107         $(window).resize(this.adjust);
108             
109         // Fire off new workspace event
110         Saiku.session.trigger('workspace:new', { workspace: this });
111 
112         return this; 
113     },
114     
115     clear: function() {
116         // Prepare the workspace for a new query
117         $(this.el).find('.workspace_results table,.connectable')
118             .html('');
119             
120         // Trigger clear event
121         Saiku.session.trigger('workspace:clear', { workspace: this });
122 
123     },
124     
125     adjust: function() {
126         // Adjust the height of the separator
127         $separator = $(this.el).find('.sidebar_separator');
128         var heightReduction = 87;
129         if (Settings.PLUGIN == true || Settings.BIPLUGIN == true) {
130             heightReduction = 2;
131         }
132         $separator.height($("body").height() - heightReduction);
133         $(this.el).find('.sidebar').height($("body").height() - heightReduction);
134         
135         // Adjust the dimensions of the results window
136         $(this.el).find('.workspace_results').css({
137             height: $("body").height() - heightReduction -
138                 $(this.el).find('.workspace_toolbar').height() - 
139                 $(this.el).find('.workspace_results_info').height() - 
140                 $(this.el).find('.workspace_editor').height() - 45
141         });
142         
143         // Fire off the adjust event
144         this.trigger('workspace:adjust', { workspace: this });
145     },
146     
147     toggle_sidebar: function() {
148         // Toggle sidebar
149         $(this.el).find('.sidebar').toggleClass('hide');
150         $(this.toolbar.el).find('.toggle_sidebar').toggleClass('on');
151         var new_margin = $(this.el).find('.sidebar').hasClass('hide') ?
152                 5 : 265;
153         $(this.el).find('.workspace_inner').css({ 'margin-left': new_margin });
154     },
155     
156     prepare: function() {
157         // Draw user's attention to cube navigation
158         $(this.el).find('.cubes')
159             .parent()
160             .css({ backgroundColor: '#AC1614' })
161             .delay(500)
162             .animate({ backgroundColor: '#fff' }, 'slow');
163     },
164     
165     new_query: function() {
166         // Delete the existing query
167         if (this.query) {
168             this.query.destroy();
169         }
170         
171         // Initialize the new query
172         this.selected_cube = $(this.el).find('.cubes').val();
173         var parsed_cube = this.selected_cube.split('/');
174         var cube = parsed_cube[3];
175         for (var i = 4; i < parsed_cube.length; i++) {
176             cube += "/" + parsed_cube[i];
177         }
178         this.query = new Query({
179             connection: parsed_cube[0],
180             catalog: parsed_cube[1],
181             schema: (parsed_cube[2] == "null" ? "" : parsed_cube[2]) ,
182             cube: decodeURIComponent(cube),
183             formatter: Settings.CELLSET_FORMATTER
184         }, {
185             workspace: this
186         });
187         
188         // Save the query to the server and init the UI
189         Saiku.session.trigger('workspace:clear', { workspace: this });
190         this.query.save();
191         this.init_query();
192     },
193     
194     init_query: function() {
195         if ((Settings.MODE == "table") && this.query) {
196             this.query.run();
197             return;
198         }
199 
200         if (this.query.get('type') == "MDX")Â {
201             this.query.set({ formatter : "flat"});
202             
203             $(this.drop_zones.el).remove();
204             this.toolbar.switch_to_mdx();
205             if (! $(this.el).find('.sidebar').hasClass('hide')) {
206                 this.toggle_sidebar();
207             }
208 
209 
210         } else {
211             $(this.el).find('.workspace_fields').show();
212             $(this.el).find('.workspace_editor .mdx_input').val('').addClass('hide');
213             $(this.toolbar.el).find('.auto, ,.toggle_fields, .query_scenario, .buckets, .non_empty, .swap_axis, .mdx, .switch_to_mdx').parent().show();
214             $(this.el).find('.run').attr('href','#run_query');
215         }
216         this.adjust();
217         if ((Settings.MODE == "view") && this.query) {
218             this.query.run();
219             return;
220         }
221 
222         // Find the selected cube
223         if (this.selected_cube === undefined) {
224             var schema = this.query.get('schema');
225             this.selected_cube = this.query.get('connection') + "/" + 
226                 this.query.get('catalog') + "/"
227                 + ((schema == "" || schema == null) ? "null" : schema) 
228                 + "/" + this.query.get('cube');
229             $(this.el).find('.cubes')
230                 .val(this.selected_cube);
231         }        
232         // Clear workspace
233         this.clear();
234         
235         if (this.selected_cube) {
236             // Create new DimensionList and MeasureList
237             this.dimension_list = new DimensionList({
238                 workspace: this,
239                 dimension: Saiku.session.sessionworkspace.dimensions[this.selected_cube]
240             });        
241             $(this.el).find('.dimension_tree').html('').append($(this.dimension_list.el));
242             
243             this.measure_list = new DimensionList({
244                 workspace: this,
245                 dimension: Saiku.session.sessionworkspace.measures[this.selected_cube]
246             });
247             $(this.el).find('.measure_tree').html('').append($(this.measure_list.el));
248         } else {
249             // Someone literally selected "Select a cube"
250             $(this.el).find('.dimension_tree').html('');
251             $(this.el).find('.measure_tree').html('');
252             return;
253         }
254 
255     },
256 
257     populate_selections: function(dimension_el) {
258 
259         if (this.other_dimension) {
260         // Populate selections - trust me, this is prettier than it was :-/
261         var axes = this.query ? this.query.get('axes') : false;
262         if (axes) {
263             for (var axis_iter = 0; axis_iter < axes.length; axis_iter++) {
264                 var axis = axes[axis_iter];
265                 var $axis = $(this.el).find('.' + 
266                     axis.name.toLowerCase() + ' ul');
267                 for (var dim_iter = 0; dim_iter < axis.dimensionSelections.length; dim_iter++) {
268                     var dimension = axis.dimensionSelections[dim_iter];
269                     var levels = [];
270                     var members = {};
271 
272                     if (dimension.name != "Measures" && dimension.selections.length > 0) {
273                         var ds = Saiku.session.sessionworkspace.dimensions[this.selected_cube].get('data');
274                         var h = dimension.selections[0].hierarchyUniqueName;
275                         _.each(ds, function(d) {
276                             if (dimension.name == d.name) {
277                                 _.each(d.hierarchies, function(hierarchy) {
278                                     if (hierarchy.uniqueName == h) {
279                                         var levels = [];
280                                         _.each(hierarchy.levels, function(level) {
281                                             levels.push(level.uniqueName);
282                                         });
283                                         dimension.selections = _.sortBy(dimension.selections, function(selection) {
284                                             return _.indexOf(levels, selection.levelUniqueName);
285                                         }); 
286                                     }
287                                 });
288                             }
289                         });
290                     } else if (dimension.name == "Measures" && dimension.selections.length > 0) {
291                         var ms = Saiku.session.sessionworkspace.measures[this.selected_cube].get('data');
292                         var mlist = [];
293                         _.each(ms, function(m) {
294                             mlist.push(m.uniqueName);
295                         });
296                         dimension.selections = _.sortBy(dimension.selections, function(selection) {
297                             return _.indexOf(mlist, selection.uniqueName);
298                         }); 
299                     }
300 
301 
302                     for (var sel_iter = 0; sel_iter < dimension.selections.length; sel_iter++) {
303                         var selection = dimension.selections[sel_iter];
304                         
305                         // Drag over dimensions and measures
306                         var type, name;
307                         if (selection.dimensionUniqueName == "Measures") {
308                             type = "measure";
309                             name = selection.uniqueName;
310                         } else {
311                             type = "dimension";
312                             name = selection.levelUniqueName;
313                         }
314                             
315                         if (levels.indexOf(name) === -1) {
316                             var $dim = $(dimension_el)
317                                 .find('a[rel="' + name + '"]')
318                                 .parent();
319                             
320                             if (!$dim.html() || $dim.html() == null) {
321                                 $dim = $(this.other_dimension)
322                                 .find('a[rel="' + name + '"]')
323                                 .parent();
324                             }
325                             var $clone = $dim.clone()
326                                 .addClass('d_' + type)
327                                 .appendTo($axis);
328                             
329                             if (type == "dimension") {
330                                 $("<span />").addClass('sprite selections')
331                                     .prependTo($clone);
332                                 $icon = $("<span />").addClass('sort');
333                                 var sort = false;
334                                 _.each(axes, function(i_axis) {
335                                     if (i_axis.sortLiteral && i_axis.sortLiteral != null && i_axis.sortLiteral.indexOf(selection.hierarchyUniqueName) != -1) {
336                                         $icon.addClass(i_axis.sortOrder);
337                                         sort = true;
338                                     }
339                                 });
340                                 if (!sort) {
341                                     $icon.addClass('none');
342                                 }
343                                 
344                                 $icon.insertBefore($clone.find('a'));
345                             }
346 
347                             if (type == "measure") {
348                                 $icon = $("<span />").addClass('sort');
349                                 var sort = false;
350                                 _.each(axes, function(i_axis) {
351                                     if (i_axis.sortLiteral && i_axis.sortLiteral != null && i_axis.sortLiteral.indexOf(name) != -1) {
352                                         $icon.addClass(i_axis.sortOrder);
353                                         sort = true;
354                                     }
355                                 });
356                                 if (!sort) {
357                                     $icon.addClass('none');
358                                 }
359                                 
360                                 $icon.insertBefore($clone.find('a'));
361                             }
362 
363                             
364                             
365                             $dim.css({fontWeight: "bold"})
366                                 .draggable('disable')
367                                 .parents('.parent_dimension')
368                                 .find('.folder_collapsed')
369                                 .css({fontWeight: "bold"}); 
370                             levels.push(name);
371                         }
372                         
373                         // FIXME - something needs to be done about selections here
374                     }
375                 }
376             }
377             
378             this.query.run(true);
379         }
380         
381         // Make sure appropriate workspace buttons are enabled
382         this.trigger('query:new', { workspace: this });
383         
384         // Update caption when saved
385         this.query.bind('query:save', this.update_caption);
386         } else {
387             this.other_dimension = dimension_el;
388         }
389 
390     },
391     
392     update_caption: function() {
393         var caption = this.query.get('name');
394         $(this.tab.el).find('a').html(caption);
395     },
396     
397    
398     
399     remove_dimension: function(event, ui) {
400         this.drop_zones.remove_dimension(event, ui);
401     }
402 });
