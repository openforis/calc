 21 /**
 22  * Dialog for member selections
 23  */
 24 var SelectionsModal = Modal.extend({
 25     type: "selections",
 26 
 27     buttons: [
 28         { text: "Save", method: "save" },
 29         { text: "Cancel", method: "close" }
 30     ],
 31     
 32     events: {
 33         'click a': 'call',
 34         'change #show_unique': 'show_unique_action',
 35         'dblclick select option' : 'click_move_selection'
 36     },    
 37 
 38     show_unique_option: false,
 39     
 40     initialize: function(args) {
 41         // Initialize properties
 42         _.extend(this, args);
 43         this.options.title = "Selections for " + this.name;
 44         this.message = "Fetching members...";
 45         this.query = args.workspace.query;
 46 
 47         _.bindAll(this, "fetch_members", "populate", "finished");
 48         // Bind selection handlers
 49         _.extend(this.events, {
 50             'click div.selection_buttons a.form_button': 'move_selection'
 51         });
 52         
 53         // Determine axis
 54         this.axis = "undefined"; 
 55         if (args.target.parents('.fields_list_body').hasClass('rows')) { 
 56             this.axis = "ROWS";
 57         }
 58         if (args.target.parents('.fields_list_body').hasClass('columns')) { 
 59             this.axis = "COLUMNS";
 60         }
 61         if (args.target.parents('.fields_list_body').hasClass('filter')) { 
 62             this.axis = "FILTER";
 63         }
 64         // Resize when rendered
 65         this.bind('open', this.post_render);
 66         this.render();
 67         
 68         $(this.el).parent().find('.ui-dialog-titlebar-close').bind('click',this.finished);
 69         // Fetch available members
 70         this.member = new Member({}, {
 71             cube: args.workspace.selected_cube,
 72             dimension: args.key
 73         });
 74         this.member.fetch({
 75             success: this.fetch_members
 76         });
 77     },
 78     
 79     fetch_members: function(model, response) {
 80         this.available_members = response;
 81 
 82         this.workspace.query.action.get("/axis/" + this.axis + "/", { 
 83             success: this.populate 
 84         });
 85     },
 86     
 87     populate: function(model, response) {
 88 
 89             // Load template
 90             $(this.el).find('.dialog_body')
 91                 .html(_.template($("#template-selections").html())(this));
 92             
 93             this.selected_members = _.detect(response, function(obj) {
 94                 return obj.name == this.member.dimension;
 95             }, this).selections;
 96             var used_members = [];
 97     
 98             // Populate both boxes
 99             $(this.el).find('.used_selections select').removeAttr('disabled');
100             var selected_members_opts = "";
101             for (var j = 0; j < this.selected_members.length; j++) {
102                 var member = this.selected_members[j];
103                 if (member.levelUniqueName == this.member.level &&
104                     member.type == "MEMBER") {
105                     selected_members_opts += "<option value='" + escape(member.uniqueName) + "'>" + member.caption + "</option>";
106                     used_members.push(member.caption);
107                 }
108             }
109             if (used_members.length > 0) {
110                 $(selected_members_opts).appendTo($(this.el).find('.used_selections select'));
111             }
112             
113             // Filter out used members
114             this.available_members = _.select(this.available_members, function(obj) {
115                 return used_members.indexOf(obj.caption) === -1;
116             });
117             
118             $(this.el).find('.available_selections select').removeAttr('disabled');
119             var available_members_opts = "";
120             for (var i = 0; i < this.available_members.length; i++) {
121                 var member = this.available_members[i];
122                 available_members_opts += "<option value='" + escape(member.uniqueName) + "'>" + member.caption + "</option>";
123             }
124             if (this.available_members.length > 0) {
125                 $(available_members_opts).appendTo($(this.el).find('.available_selections select'));
126             }
127             var self = this;
128             $(this.el).find('.filterbox').autocomplete({
129                     minLength: 1,
130                     source: function(request, response ) {
131                         response( $.map( self.available_members, function( item ) {
132                                         if (item.caption.toLowerCase().indexOf(request.term.toLowerCase()) > -1) {
133                                             return {
134                                                 label: item.caption ,
135                                                 value: item.uniqueName
136                                             };
137                                         }
138                                 }));
139                     },
140                     select:  function(event, ui) { 
141                         var value = self.show_unique_option == false? escape(ui.item.value) : ui.item.label;
142                         $(self.el).find('.available_selections select option[value="' + value + '"]')
143                             .appendTo($(self.el).find('.used_selections select'));
144                         $('#filter_selections').val('');
145 
146                     }, close: function(event, ui) { 
147                         $('#filter_selections').val('');
148                     }
149 
150 
151                 }).data( "autocomplete" )._renderItem = function( ul, item ) {
152                 return $( "<li></li>" )
153                     .data( "item.autocomplete", item )
154                     .append( "<a class='label'>" + item.label + "</a><br><a class='description'>" + item.value + "</a>" )
155                     .appendTo( ul );
156                 };
157 		// Translate
158 		Saiku.i18n.translate();
159         // Show dialog
160         Saiku.ui.unblock();
161     },
162     
163     post_render: function(args) {
164         $(args.modal.el).parents('.ui-dialog').css({ width: "800px", left: "inherit", margin:"0 auto" });
165     },
166     
167     move_selection: function(event) {
168         var action = $(event.target).attr('id');
169         var $to = action.indexOf('add') !== -1 ? 
170             $(this.el).find('.used_selections select') :
171             $(this.el).find('.available_selections select');
172         var $from = action.indexOf('add') !== -1 ? 
173             $(this.el).find('.available_selections select') :
174             $(this.el).find('.used_selections select');
175         var $els = action.indexOf('all') !== -1 ? 
176             $from.find('option') :$from.find('option:selected');
177         $els.detach().appendTo($to);
178     },
179 
180     click_move_selection: function(event, ui) {
181       var to = ($(event.target).parent().parent().hasClass('used_selections')) ? '.available_selections' : '.used_selections';
182       $(event.target).appendTo($(this.el).find(to +' select'));
183     },
184 
185     show_unique_action: function() {
186         $.each($(this.el).find('option'), function(i, option) {
187             var text = $(option).text();
188             $(option).text(unescape($(option).val()));
189             $(option).val(escape(text));
190         });
191         this.show_unique_option= ! this.show_unique_option;
192     },
193     
194     save: function() {
195         // Notify user that updates are in progress
196         var $loading = $("<div>Saving...</div>");
197         $(this.el).find('.dialog_body').children().hide();
198         $(this.el).find('.dialog_body').prepend($loading);
199         var show_u = this.show_unique_option;
200 
201         // Determine updates
202         var updates = [{
203             hierarchy: this.member.hierarchy,
204             uniquename: this.member.level,
205             type: 'level',
206             action: 'delete'
207         }];
208         
209         // If no selections are used, add level
210         if ($(this.el).find('.used_selections option').length === 0) {
211             updates.push({
212                 hierarchy: this.member.hierarchy,
213                 uniquename: this.member.level,
214                 type: 'level',
215                 action: 'add'
216             });
217         } else {
218             // Loop through selections
219             $(this.el).find('.used_selections option')
220                 .each(function(i, selection) {
221                 var value = show_u ? 
222                     $(selection).text() : unescape($(selection).val());
223                 updates.push({
224                     uniquename: value,
225                     type: 'member',
226                     action: 'add'
227                 });
228             });
229         }
230         
231         // Notify server
232         this.query.action.put('/axis/' + this.axis + '/dimension/' + this.member.dimension, { 
233             success: this.finished,
234             data: {
235                 selections: JSON.stringify(updates)
236             }
237         });
238         
239         return false;
240     },
241     
242     finished: function() {
243         $('#filter_selections').remove();
244         this.available_members = null;
245         $(this.el).dialog('destroy').remove();
246         this.query.run();
247     }
248 });
