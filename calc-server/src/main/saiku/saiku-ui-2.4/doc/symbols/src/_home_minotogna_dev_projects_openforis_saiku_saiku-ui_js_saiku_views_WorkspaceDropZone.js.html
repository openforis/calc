 22 /**
 23  * Sets up workspace drop zones for DnD and other interaction
 24  */
 25 var WorkspaceDropZone = Backbone.View.extend({
 26     template: function() {
 27         var template = $("#template-workspace-dropzones").html() || "";
 28         return _.template(template)();
 29     },
 30     
 31     events: {
 32         'sortbeforestop': 'select_dimension',
 33         'click .d_dimension span.selections': 'selections',
 34         'click .d_dimension a': 'selections',
 35         'click .d_measure a' : 'remove_dimension',
 36         'click .d_measure span.sort' : 'sort_measure',
 37         'click .d_dimension span.sort' : 'sort_measure'
 38     },
 39     
 40     initialize: function(args) {
 41         // Keep track of parent workspace
 42         this.workspace = args.workspace;
 43         
 44         // Maintain `this` in jQuery event handlers
 45         _.bindAll(this, "select_dimension", "move_dimension", 
 46                 "remove_dimension", "update_selections","sort_measure");
 47     },
 48     
 49     render: function() {
 50         // Generate drop zones from template
 51         $(this.el).html(this.template());
 52         
 53         // Activate drop zones
 54         $(this.el).find('.connectable').sortable({
 55             connectWith: $(this.el).find('.connectable'),
 56             cursorAt: {
 57                 top: 10,
 58                 left: 35
 59             },
 60             forcePlaceholderSize: true,
 61             items: '> li',
 62             opacity: 0.60,
 63             placeholder: 'placeholder',
 64             tolerance: 'pointer',
 65             
 66             start: function(event, ui) {
 67                 ui.placeholder.text(ui.helper.text());
 68             }
 69 
 70         });
 71         
 72         return this; 
 73     },
 74 
 75     sort_measure: function(event, ui) {
 76         $axis = $(event.target).parent().parents('.fields_list_body');
 77         var source = "";
 78         var target = "ROWS";
 79         
 80         if ($axis.hasClass('rows')) { source = "ROWS"; target= "COLUMNS"; }
 81         if ($axis.hasClass('columns')) { source = "COLUMNS"; target="ROWS"; }
 82         if ($axis.hasClass('filter')) { source = "FILTER"; target="COLUMNS"; }
 83 
 84         var sortOrder = "";
 85         if ($(event.target).hasClass('none')) sortOrder = "none";
 86         if ($(event.target).hasClass('BASC')) sortOrder = "BASC";
 87         if ($(event.target).hasClass('BDESC')) sortOrder = "BDESC";
 88 
 89         var memberpath = $(event.target).parent().find('a').attr('href').replace('#', '').split('/');
 90         var member = "-";
 91         if ($(event.target).parent().hasClass('d_dimension')) {
 92             member = memberpath[2] + ".CurrentMember.Name";
 93             $axis.find('.d_dimension .sort').removeClass('BASC').removeClass('BDESC').addClass('none');
 94             $axis.parent().parent().find("." + target.toLowerCase() + " .d_measure .sort").removeClass('BASC').removeClass('BDESC').addClass('none');
 95             target = source;
 96         } else {
 97             member = memberpath[memberpath.length -1];
 98             $axis.find('.d_measure .sort').removeClass('BASC').removeClass('BDESC').addClass('none');
 99             $axis.parent().parent().find("." + target.toLowerCase() + " .d_dimension .sort").removeClass('BASC').removeClass('BDESC').addClass('none');
100         }
101 
102         var futureSortOrder = "none";
103         if (sortOrder == "none") futureSortOrder = "BASC";
104         if (sortOrder == "BASC") futureSortOrder = "BDESC";
105         if (sortOrder == "BDESC") futureSortOrder = "none";
106         
107 
108 
109         $(event.target).removeClass('none').addClass(futureSortOrder);
110 
111         if (futureSortOrder == "none") {
112             var url = "/axis/" + target + "/sort/";
113             this.workspace.query.action.del(url, {
114                 success: this.workspace.query.run
115             });
116         } else {
117             var url = "/axis/" + target + "/sort/" + futureSortOrder + "/" + member;
118             this.workspace.query.action.post(url, {
119                 success: this.workspace.query.run
120             });
121         }
122 
123     },
124     
125     select_dimension: function(event, ui) {
126 
127         $axis = ui.item.parents('.fields_list_body');
128         var target = "";
129         
130         if ($axis.hasClass('rows')) target = "ROWS";
131         if ($axis.hasClass('columns')) target = "COLUMNS";
132         if ($axis.hasClass('filter')) target = "FILTER";
133 
134 
135         // Short circuit if this is a move
136         if (ui.item.hasClass('d_measure') ||
137                 ui.item.hasClass('d_dimension')) {
138             this.move_dimension(event, ui, target);
139             return;
140         }
141 
142         // Make the element and its parent bold
143         var original_href = ui.item.find('a').attr('href');
144         var $original = $(this.workspace.el).find('.sidebar')
145             .find('a[href="' + original_href + '"]').parent('li');
146         $original
147             .css({fontWeight: "bold"})
148             .draggable('disable');
149         $original.parents('.parent_dimension')
150             .find('.folder_collapsed')
151             .css({fontWeight: "bold"});
152         
153 
154         // Wrap with the appropriate parent element
155         if (ui.item.find('a').hasClass('level')) {
156             var $icon = $("<div />").addClass('sprite').addClass('selections');
157             var $icon2 = $("<span />").addClass('sprite').addClass('sort none');
158         
159             ui.item.addClass('d_dimension').prepend($icon);
160             ui.item.addClass('d_dimension').prepend($icon2);
161         } else {
162             var $icon = $("<span />").addClass('sort none');
163             ui.item.addClass('d_measure').prepend($icon);
164         }
165 
166 
167 
168         var member = ui.item.find('a').attr('href').replace('#', '');
169         var dimension = member.split('/')[0];
170         var dimensions = [];
171 
172         this.update_selections(event,ui);
173 
174         $axis.find('a').each( function(i,element) {
175             var imember = $(element).attr('href');
176             var idimension = imember.replace('#', '').split('/')[0]; 
177             if (dimensions.indexOf(idimension) == -1) {
178                 dimensions.push(idimension);
179             }
180         });
181 
182         var index = dimensions.indexOf(dimension);
183 
184 
185         // Notify the model of the change
186         this.workspace.query.move_dimension(member, 
187                 target, index);
188 
189         if ("FILTER" == target) {
190             var ev = { target : $axis.find('a[href="#' + member + '"]') };
191             this.selections(ev, ui);
192 
193         }
194 
195         // Prevent workspace from getting this event
196         return true;
197     },
198     
199     move_dimension: function(event, ui, target) {
200         if (! ui.item.hasClass('deleted')) {
201             $axis = ui.item.parents('.fields_list_body');
202 
203             // Notify the model of the change
204             var dimension = ui.item.find('a').attr('href').replace('#', '').split('/')[0];
205             var dimensions = [];
206 
207             this.update_selections(event,ui);
208 
209             $axis.find('a').each( function(i,element) {
210                 var imember = $(element).attr('href');
211                 var idimension = imember.replace('#', '').split('/')[0]; 
212                 if (dimensions.indexOf(idimension) == -1) {
213                     dimensions.push(idimension);
214                 }
215             });
216             var index = dimensions.indexOf(dimension);
217 
218             
219                 this.workspace.query.move_dimension(dimension, 
220                     target, index);
221         }
222         
223         // Prevent workspace from getting this event
224         event.stopPropagation();
225         return false;
226     },
227 
228     update_selections: function(event, ui) {
229         var member = ui.item.find('a').attr('href');
230         var dimension = member.replace('#', '').split('/')[0];
231         var index = ui.item.parent('.connectable').children().index(ui.item);
232         var axis = ui.item.parents('.fields_list_body');
233         var allAxes = axis.parent().parent();
234         var target = '';
235         var source = '';
236         var myself = this;
237         var $originalItem =  $(myself.workspace.el).find('.sidebar')
238                                     .find('a[href="' + member + '"]').parent();
239         var insertElement = $(ui.item);
240         var type = $(ui.item).hasClass('d_dimension') ? "d_dimension" : "d_measure";
241 
242         var sourceAxis = "";
243         if ($axis.hasClass('rows')) sourceAxis = "ROWS";
244         if ($axis.hasClass('columns')) sourceAxis = "COLUMNS";
245         if ($axis.hasClass('filter')) sourceAxis = "FILTER";
246 
247         source = ".rows, .columns, .filter";
248         allAxes.find(source).find('a').each( function(index, element) {
249             var p_member = $(element).attr('href').replace('#', '');
250             var p_dimension = p_member.split('/')[0];
251             if (p_dimension == dimension && (( "#" + p_member) != member )) {
252                 $(element).parent().remove();
253             }
254         });
255         
256         var n_dimension = null;
257         var p_dimension = null;
258 
259         
260         var prev = $(ui.item).prev();
261         if (prev && prev.length > 0) {
262             var p_member = prev.find('a').attr('href');
263             p_dimension = p_member.replace('#', '').split('/')[0];
264         }
265         var next = $(ui.placeholder).next();
266 
267         while (p_dimension != null && next && next.length > 0 ) {
268             var n_member = next.find('a').attr('href');
269             n_dimension = n_member.replace('#', '').split('/')[0]; 
270             if (p_dimension == n_dimension) {
271                 next.insertBefore($(ui.item));
272             } else {
273                 p_dimension = null;
274             }
275             next = $(ui.placeholder).next();
276         }
277             
278 
279         $originalItem.parent().find('.ui-draggable-disabled').clone().attr('class', 'ui-draggable').removeAttr('style')
280                                 .addClass(type)
281                                 .insertAfter(insertElement);
282         
283 
284         axis.find('.d_dimension a').each( function(index, element) {
285             element = $(element);
286             if (!element.prev() || (element.prev() && element.prev().length == 0)) {
287                 var $icon = $("<span />").addClass('sprite').addClass('selections');
288                 $icon.insertBefore(element);
289                 var $icon = $("<span />").addClass('sprite').addClass('sort none');
290                 $icon.insertBefore(element);
291             }
292         });
293 
294         axis.find('.d_measure a, .d_dimension a').each( function(index, element) {
295             element = $(element);
296             if (!element.prev() || (element.prev() && element.prev().length == 0)) {
297                 if (sourceAxis != "FILTER") {
298                     var $icon = $("<span />").addClass('sort none');
299                     $icon.insertBefore(element);
300                 }
301                 
302             }
303         });
304 
305         $(ui.item).remove();
306 
307     },
308 
309 
310     remove_dimension: function(event, ui) {
311         // Reenable original element
312         var $source = ui ? ui.draggable : $(event.target).parent();
313         var original_href = $source.find('a').attr('href');
314         var $original = $(this.workspace.el).find('.sidebar')
315             .find('a[href="' + original_href + '"]').parent('li');
316         $original
317             .draggable('enable')
318             .css({ fontWeight: 'normal' });
319         
320         // Unhighlight the parent if applicable
321         if ($original.parents('.parent_dimension')
322                 .children().children('.ui-state-disabled').length === 0) {
323             $original.parents('.parent_dimension')
324                 .find('.folder_collapsed')
325                 .css({fontWeight: "normal"});
326         }
327         
328         // Notify server
329         var target = '';
330         var dimension = original_href.replace('#', '');
331         $target_el = $source.parent().parent('div.fields_list_body');
332         if ($target_el.hasClass('rows')) target = "ROWS";
333         if ($target_el.hasClass('columns')) target = "COLUMNS";
334         if ($target_el.hasClass('filter')) target = "FILTER";
335         
336         var url = "/axis/" + target + "/dimension/" + dimension;
337         this.workspace.query.action.del(url, {
338             success: this.workspace.query.run
339         });
340         
341         // Remove element
342         $source.addClass('deleted').remove();
343         
344         // Prevent workspace from getting this event
345         event.stopPropagation();
346         event.preventDefault();
347         return false;
348     },
349     
350     selections: function(event, ui) {
351         // Determine dimension
352         var $target = $(event.target).hasClass('sprite') ?
353             $(event.target).parent().find('.level') :
354             $(event.target);
355         var key = $target.attr('href').replace('#', '');
356         
357         // Launch selections dialog
358         (new SelectionsModal({
359             target: $target,
360             name: $target.text(),
361             key: key,
362             workspace: this.workspace
363         })).open();
364         
365         // Prevent default action
366         try {
367             event.preventDefault();
368         } catch (e) { }
369         return false;
370     }
371 });
